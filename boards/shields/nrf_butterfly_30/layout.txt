# nrf_butterfly_30 keyboard layout — source of truth for the keymap.
#
# Edit this file, then run generate-keymap.ps1 in this same folder to
# regenerate nrf_butterfly_30.keymap. Do not hand-edit the .keymap file -- it
# gets overwritten.
#
# Each layer is a 3-row x 10-col grid, in the same left-to-right,
# top-to-bottom physical order you'd read the keyboard. The "|" is just a
# divider for readability -- the generator ignores it, so exact spacing
# doesn't matter.
#
# Layer order in this file IS the layer index (0-based) used by MO/TOG/TO and
# the TSYM/HNUM hold-taps below. Currently: 0=base, 1=l2, 2=num, 3=sym.
#
# ZMK resolves a key from the highest-numbered *active* layer that isn't
# transparent there, so this order gives priority sym > num > l2 > base,
# matching the original Arduino firmware.
#
# Token reference:
#   A-Z, 0-9                    a letter or digit key
#   ESC TAB BSPC DEL ENTER SPACE      named keys
#   LSFT RSFT LCTL RCTL LALT LGUI     modifiers
#   GRAVE MINUS EQUAL LBKT RBKT BSLH SEMI SQT COMMA DOT FSLH COLON   punctuation
#   TRNS                              transparent -- falls through to the
#                                      next active layer below this one
#   NONE                              key does nothing
#   MO<n>                             hold: momentary-activate layer n
#   TOG<n>                            press: toggle layer n on/off
#   TO<n>                             press: switch to layer n only,
#                                      deactivating every other layer
#   TSYM                              tap = t, hold = momentary sym layer (3)
#   HNUM                              tap = h, hold = momentary num layer (2)
#   BOOT                              reboot into bootloader (for reflashing)
#   BT0 BT1 BT2 BT3 BT4               select Bluetooth profile 0-4
#   OUTTOG                            toggle USB/BLE output
#   BTCLR                             clear current Bluetooth profile's bond

[base]
Q     W     F     P     G     | J     L     U     Y     TOG1
A     R     S     TSYM  D     | HNUM  N     E     I     O
Z     X     C     V     B     | SPACE K     M     COMMA DOT

[l2]
TRNS  TRNS  TRNS  TRNS  TRNS  | TRNS  TRNS  TRNS  TRNS  TRNS
TRNS  TRNS  TRNS  TRNS  TRNS  | TRNS  TRNS  TRNS  TRNS  TRNS
TRNS  TRNS  TRNS  TRNS  TRNS  | TRNS  TRNS  TRNS  LCTL  LSFT

TRNS  TRNS  TRNS  TRNS  TRNS  | TRNS  TRNS  TRNS  TRNS  TRNS
